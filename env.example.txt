# Supabase (PostgreSQL) 설정
# 이 파일을 복사해서 .env 파일을 만들고, 실제 값으로 변경하세요

# Supabase Project URL
# Supabase Dashboard > Settings > API > Project URL
EXPO_PUBLIC_SUPABASE_URL=https://your-project.supabase.co

# Supabase Anon Key
# Supabase Dashboard > Settings > API > Project API keys > anon public
EXPO_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here

# Google Gemini API (선택사항 - AI 영양 정보 자동 입력)
# Google AI Studio > https://aistudio.google.com/app/apikey
EXPO_PUBLIC_GEMINI_API_KEY=your-gemini-api-key-here

# ===== AI 이미지 생성 (선택사항) =====

# AI 이미지 생성 활성화 여부
# false: Unsplash + 사전 정의 이미지 사용 (무료, 빠름)
# true: AI로 이미지 생성 (유료/느림, 하지만 정확함)
EXPO_PUBLIC_ENABLE_AI_IMAGE_GEN=false

# OpenAI API Key (DALL-E 3 이미지 생성)
# https://platform.openai.com/api-keys
# 비용: $0.04/image (1024x1024 standard)
# 신규 계정: $5 무료 크레딧 (125장)
EXPO_PUBLIC_OPENAI_API_KEY=your-openai-api-key-here

# Unsplash API (선택사항 - 실제 음식 사진 검색)
# Unsplash Developers > https://unsplash.com/developers
# 무료: 50 requests/hour
EXPO_PUBLIC_UNSPLASH_ACCESS_KEY=your-unsplash-access-key-here

# ===== 소셜 로그인 (선택사항) =====

# 네이버 로그인
# 네이버 개발자 센터 > https://developers.naver.com/apps/#/register
# 애플리케이션 등록 후 Client ID/Secret 발급
EXPO_PUBLIC_NAVER_CLIENT_ID=your-naver-client-id-here
EXPO_PUBLIC_NAVER_CLIENT_SECRET=your-naver-client-secret-here

# ===== 개발 환경 설정 =====

# 개발 모드 (development, production)
# development: 이메일 확인 없이 즉시 로그인 가능 (Supabase 설정 필요)
NODE_ENV=development

# ⚠️ 주의사항:
# 1. .env 파일은 절대 Git에 커밋하지 마세요 (이미 .gitignore에 포함됨)
# 2. service_role 키는 사용하지 마세요 (보안 위험)
# 3. .env 파일 수정 후 앱을 재시작하세요
# 4. Gemini API와 Unsplash API는 선택사항이며, 없어도 앱은 정상 작동합니다
# 5. Unsplash API 키가 없으면 기본 이미지가 사용됩니다
# 6. 네이버 Client Secret은 서버에서만 사용해야 하므로 주의하세요!
